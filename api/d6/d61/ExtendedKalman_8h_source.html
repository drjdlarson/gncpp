70 matrix::Vector<T> ExtendedKalman<T>::predict(
71  T timestep, const matrix::Vector<T>& curState,

73  const BayesPredictParams* params) {

77  }

92  this->dynamicsModel())
93  ->getStateMat(timestep, params->stateTransParams.get());
94  } else {
95  throw exceptions::TypeError("Unknown dynamics type");
96  }
97 
98  if (m_continuousCov) {
99  throw std::runtime_error(
100  "Continuous covariance model not implemented yet for the EKF");
101  } else {
102  this->cov =
103  stateMat * this->cov * stateMat.transpose() + this->m_procNoise;
104  }
105

109 template <typename T>
110 matrix::Vector<T> ExtendedKalman<T>::correct([[maybe_unused]] T timestep,
111  const matrix::Vector<T>& meas,
112  const matrix::Vector<T>& curState,
113  T& measFitProb,
114  const BayesCorrectParams* params) {
115  if (params != nullptr && !utilities:: instanceof
116  <BayesCorrectParams>(params)) {
117  throw exceptions::BadParams("Params must be BayesCorrectParams");
118  }
119 
120  matrix::Vector<T> estMeas =
121  this->measurementModel()->measure(curState, params->measParams.get());
122  matrix::Matrix<T> measMat = this->measurementModel()->getMeasMat(
123  curState, params->measParams.get());
124 
125  matrix::Matrix<T> inovCov = measMat * this->cov * measMat.transpose();
126 
127  matrix::Matrix<T> kalmanGain =
128  this->cov * measMat.transpose() * inovCov.inverse();
129 
130  matrix::Vector<T> inov = meas - estMeas;
131  this->cov -= kalmanGain * measMat * this->cov;
132 
133  measFitProb = math::calcGaussianPDF(meas, estMeas, inovCov);
134 
135  return curState + kalmanGain * inov;
136 }
